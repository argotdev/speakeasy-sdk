# RunObject

Represents an execution run on a [thread](/docs/api-reference/threads).

## Example Usage

```typescript
import { RunObject } from "argot-open-ai/models/components";

let value: RunObject = {
  id: "<id>",
  object: "thread.run",
  createdAt: 360545,
  threadId: "<id>",
  assistantId: "<id>",
  status: "incomplete",
  requiredAction: {
    type: "submit_tool_outputs",
    submitToolOutputs: {
      toolCalls: [
        {
          id: "<id>",
          type: "function",
          function: {
            name: "<value>",
            arguments: "<value>",
          },
        },
      ],
    },
  },
  lastError: {
    code: "invalid_prompt",
    message: "<value>",
  },
  expiresAt: 46007,
  startedAt: 232627,
  cancelledAt: 348519,
  failedAt: 814967,
  completedAt: 985492,
  incompleteDetails: {},
  model: "XC90",
  instructions: "<value>",
  tools: [
    {
      type: "function",
      function: {
        name: "<value>",
      },
    },
  ],
  metadata: {},
  usage: {
    completionTokens: 296556,
    promptTokens: 992012,
    totalTokens: 249420,
  },
  maxPromptTokens: 105906,
  maxCompletionTokens: 950953,
  truncationStrategy: {
    type: "auto",
  },
  toolChoice: {
    type: "function",
  },
  responseFormat: {
    type: "json_object",
  },
};
```
